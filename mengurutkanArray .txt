 fun main(args: Array<String>) {            val arrayNumber = intArrayOf(4, 9, 7, 5, 8, 9, 3)            println("Array sesuai urutan aslinya :")            for (i in arrayNumber.indices) print(arrayNumber[i].toString() + "  ")            mengurutkanElemen(arrayNumber)            println("\n\nArray setelah diurutkan :")            for (j in arrayNumber.indices) print(arrayNumber[j].toString() + "  ")        }         fun mengurutkanElemen(array: IntArray) {            for (i in array.size - 1 downTo 1) {                for (j in 0..i) {                    if (array[j] > array[i]) menukar(array, j, i)                }            }        }        fun menukar(arrayB: IntArray, indexj: Int, indexi: Int) {            val sementara: Int = arrayB[indexi]            arrayB[indexi] = arrayB[indexj]            arrayB[indexj] = sementara        }
